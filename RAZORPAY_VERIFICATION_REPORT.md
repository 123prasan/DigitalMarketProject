## Code Snippet Comparison

### Snippet 1: Initialize Payment Gateway

```javascript
// BEFORE - Cashfree
const { Cashfree } = require("cashfree-pg");
Cashfree.XClientId = process.env.CASHFREE_APP_ID;
Cashfree.XClientSecret = process.env.CASHFREE_SECRET_KEY;
Cashfree.XEnvironment = Cashfree.Environment.SANDBOX;

// AFTER - Razorpay
const Razorpay = require("razorpay");
const razorpayInstance = new Razorpay({
  key_id: process.env.RAZORPAY_KEY_ID,
  key_secret: process.env.RAZORPAY_KEY_SECRET,
});
```

### Snippet 2: Create Order

```javascript
// BEFORE - Cashfree
const response = await Cashfree.PGCreateOrder("2023-08-01", {
  order_id: `order_${fileId}_${Date.now()}`,
  order_amount: amountInRupees,
  order_currency: "INR",
  customer_details: { /* ... */ },
  order_meta: {
    return_url: `${process.env.BASE_URL}/payment-success?order_id={order_id}`,

const response = await razorpayInstance.orders.create({
```

### Snippet 3: Verify Payment Signature

```javascript
// BEFORE - Cashfree
const expectedSignature = crypto
  .createHmac('sha256', secretKey)
  .update(JSON.stringify(req.body))
  .digest('base64');

if (signature !== expectedSignature) {
  return res.status(400).json({ error: 'Invalid signature' });
}

// AFTER - Razorpay
const expectedSignature = crypto
  .createHmac('sha256', process.env.RAZORPAY_KEY_SECRET)
  .update(razorpay_order_id + "|" + razorpay_payment_id)
  .digest('hex');

if (signature !== expectedSignature) {
  return res.status(400).json({ error: 'Invalid signature' });
}
```

### Snippet 4: Fetch Payment Details

```javascript
// BEFORE - Cashfree
try {
  const response = await Cashfree.PGFetchOrder("2023-08-01", order_id);
  const paymentDetails = response.data;
  
  if (paymentDetails.order_status !== 'ACTIVE') {
    return res.status(400).json({ success: false, message: "Payment not completed" });
  }
} catch (err) {
  return res.status(502).json({ success: false, message: "Payment gateway error" });
}

// AFTER - Razorpay
try {
  const paymentDetails = await razorpayInstance.payments.fetch(razorpay_payment_id);
  
  if (paymentDetails.status !== 'captured') {
    return res.status(400).json({ success: false, message: "Payment not completed" });
  }
} catch (err) {
  return res.status(502).json({ success: false, message: "Payment gateway error" });
}
```

### Snippet 5: Frontend Checkout

```javascript
// BEFORE - Cashfree
const cashfree = new Cashfree({mode:"sandbox"});
cashfree.checkout({
    paymentSessionId: order.payment_session_id,
    redirectTarget: "_modal",
}).then((result) => {
    if (result.error) {
        alert(result.error.message);
    } else if (result.paymentDetails) {
        fetch('/verify-payment', {
            method: 'POST',
            body: JSON.stringify({
              order_id: result.paymentDetails.orderId,
              payment_id: result.paymentDetails.paymentId,
              // ...
            })
        });
    }
});

// AFTER - Razorpay
const options = {
    key: order.key,
    amount: order.amount,
    currency: order.currency,
    name: "Vidyari",
    description: file.filename,
    order_id: order.order_id,
    handler: function (response) {
        fetch('/verify-payment', {
            method: 'POST',
            body: JSON.stringify({
              razorpay_order_id: response.razorpay_order_id,
              razorpay_payment_id: response.razorpay_payment_id,
              razorpay_signature: response.razorpay_signature,
              // ...
            })
        });
    },
    theme: { color: "#000000" }
};
const rzp = new Razorpay(options);
rzp.open();
```
